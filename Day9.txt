# search for first occurance of an element with duplicated
def search_element_dup(nums, target):
    l = 0
    r = len(nums) - 1
    while l <= r:
        mid = l+(r-l) // 2
        if nums[mid] == target:
            res = mid
            r = mid - 1
        elif nums[mid] < target:
            l = mid + 1
        else:
            r = mid - 1
    return res

nums = [1, 2, 2, 2, 3, 4, 5, 5]
target = 5
print(search_element_dup(nums, target))

# 704. Binary Search
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        l = 0
        r = len(nums) - 1
        while l <= r:
            mid = l+(r-l) // 2
            if nums[mid] == target:
                return mid
            elif nums[mid] < target:
                l = mid + 1
            else:
                r = mid - 1
        return -1


# 1672. Richest Customer Wealth
class Solution:
    def maximumWealth(self, accounts: List[List[int]]) -> int:
        res = 0
        for i in accounts:
            total = 0
            for j in i:
                total += j
            res = max(res, total)
        return res

# 69. Sqrt(x)
class Solution:
    def mySqrt(self, x: int) -> int:
        l = 0
        r = x
        res = 0
        while l <= r:
            mid = l + (r - l) // 2
            if mid * mid == x:
                return mid
            elif mid * mid < x:
                res = mid
                l = mid + 1
            else:
                r = mid - 1

        return res

# 875. Koko Eating Bananas
class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:

        l = 1
        r = max(piles)
        while l < r:
            mid = l + (r - l) // 2
            hours = 0
            for p in piles:
                hours += (p + mid - 1) // mid
            if hours <= h:
                r = mid
            else:
                l = mid + 1
        return l

# 153. Find Minimum in Rotated Sorted Array
class Solution:
    def findMin(self, nums: List[int]) -> int:
        l = 0
        r = len(nums) - 1

        while l < r:
            mid = l + (r - l) // 2
            if nums[mid] > nums[r]:
                l = mid + 1
            else:
                r = mid
        return nums[l]

# 367. Valid Perfect Square
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        l = 1
        r = num

        while l <= r:
            mid = l + (r - l) // 2
            s = mid * mid
            if s == num:
                return True
            elif s < num:
                l = mid + 1
            else:
                r = mid - 1
        return False

# 744. Find Smallest Letter Greater Than Target
class Solution(object):
    def nextGreatestLetter(self, letters, target):

        l = 0
        r = len(letters) - 1
        while l <= r:
            mid = l + (r - l) // 2
            if letters[mid] <= target:
                l = mid + 1
            else:
                r = mid - 1
        return letters[l % len(letters)]

# 74. Search a 2D Matrix
class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        m = len(matrix)
        n = len(matrix[0])
        l = 0
        r = m * n - 1
        while l <= r:
            mid = l + (r - l) // 2
            row = mid // n
            cols = mid % n

            if matrix[row][cols] == target:
                return True
            elif matrix[row][cols] < target:
                l = mid + 1
            else:
                r = mid - 1

        return False


# Bubble sort
def bubbleSort(arr):
        n = len(arr)
        for i in range(n):
            for j in range(n - i - 1):
                if arr[j] > arr[j + 1]:
                    arr[j], arr[j + 1] = arr[j + 1], arr[j]
        return arr
print(bubbleSort([9,8,7,6,5,4,3,2,1]))

# selection sort
def selectionSort(arr):
    n = len(arr)
    for i in range(n):
        mini = i
        for j in range(i + 1, n):
            if arr[j] < arr[mini]:
                mini = j
        arr[i], arr[mini] = arr[mini], arr[i]
    return arr
print(selectionSort([9,8,7,6,5,4,3,2,1]))

# merge sort
def mergeSort(arr):
    if len(arr) <= 1:
        return arr
    mid = len(arr) // 2
    left = mergeSort(arr[:mid])
    right = mergeSort(arr[mid:])
    return merge(left, right)

def merge(left, right):
    i = 0
    j = 0
    ans = []
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            ans.append(left[i])
            i += 1
        else:
            ans.append(right[j])
            j += 1

    while i < len(left):
        ans.append(left[i])
        i += 1
    while j < len(right):
        ans.append(right[j])
        j += 1

    return ans
print(mergeSort([5,4,3,2,1]))

# 134. Gas Station
class Solution(object):
    def canCompleteCircuit(self, gas, cost):
        if sum(gas) < sum(cost):
            return -1
        start = 0
        tank = 0
        for i in range(len(gas)):
            tank += gas[i] - cost[i]
            if tank < 0:
                start = i + 1
                tank = 0
        return start
